#type conversions

package first;

public class autoboxing {
  public static void main(String args[]) {
    int a=10;
    Integer obj=a;
    System.out.println("Autoboxing"+obj);
    
    Integer ob1=20;
    int b=ob1;
    System.out.println("Unboxing "+b);
    
  }

}
